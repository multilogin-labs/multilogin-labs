```javascript
const MULTILOGIN_API = "https://api.multilogin.com";

async function launchProfile(profileId) {
  const res = await fetch(`${MULTILOGIN_API}/v2/profile/start`, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${process.env.MULTILOGIN_TOKEN}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({ profileId }),
  });
  const { port } = await res.json();
  console.log(`Profile on port ${port}`);
}
```
